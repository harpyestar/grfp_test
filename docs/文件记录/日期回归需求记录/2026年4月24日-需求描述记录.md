1. 首先到/home页

4. 点击搜索栏目的Project填入固定项目名：hyg-自动化项目-未完成项

def run(playwright: Playwright) -> None:
    browser = playwright.chromium.launch(headless=False)
    context = browser.new_context()
    page.get_by_role("tab", name="Started", exact=True).click()
    page.locator("div").filter(has_text=re.compile(r"^Project$")).nth(1).click()
    page.locator("#el-id-1929-57").fill("hyg-自动化项目-未完成项")
    page.locator(".search > .btn").click()
    page.locator("div").filter(has_text=re.compile(r"^Start$")).click()
