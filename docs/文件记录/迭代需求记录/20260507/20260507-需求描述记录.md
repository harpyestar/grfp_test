def run(playwright: Playwright) -> None:
    browser = playwright.chromium.launch(headless=False)
    context = browser.new_context()
    page.get_by_role("button", name="RFP Management").click()
    page.get_by_role("menuitem", name="Contracting").click()
    page.get_by_role("tab", name="Started", exact=True).click()
    page.locator("#el-id-1615-59").click()
    page.locator("#el-id-1615-59").fill("hyg-测试项目-新加坡酒店类")
    page.locator(".search > .btn").click()
    page.get_by_text("Contract Signing").first.click()
    page.get_by_text("Mondrian Singapore Duxton Four Star16A DUXTON HILL, Singapore, 089970, Duxton").click()
    with page.expect_popup() as page1_info:
        page.get_by_role("button", name="Detail").click()
    page1 = page1_info.value
    page1.get_by_role("button", name="内部备注").click()
    page1.get_by_placeholder("请输入内部跟进备注").click()
    page1.get_by_placeholder("请输入内部跟进备注").fill("hyhy")
    page1.get_by_role("button", name="确定").click()

    # ---------------------
    context.close()
    browser.close()
